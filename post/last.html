# Quick port scanner — Python
import socket

def scan(host, ports):
    for port in ports:
        s = socket.socket()
        s.settimeout(0.5)
        result = s.connect_ex((host, port))
        if result == 0:
            print(f"[+] {host}:{port} — OPEN")
        s.close()

scan("10.10.10.1", range(1, 1025))
